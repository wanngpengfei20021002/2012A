## Weather 天气组件

```javascript
    import Weather from ''
    
    /**
		 * 组件参数
     * 
     * @param {province} { string } 城市
     * @param {city} { string } 城区
     * @param {weather} { string } 天气
     * @param {temperature} { string } 温度
     * @param {winddirection} { string } 风向
     * @param {reporttime} { string } 日期
    */

    // 天气组件
    <Weather
      province={province} // 城市
      city={city} // 城区
      weather={weather} // 必填表单
      temperature={temperature} // 温度
      winddirection={winddirection} // 风向
      reporttime={reporttime} // 日期
    />

```
